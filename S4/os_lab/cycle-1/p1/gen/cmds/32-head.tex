\subsection{Command 32: head} 
\textbf{Purpose}
\begin{flushleft}
 display first lines of a file
\end{flushleft}
\textbf{Usage}
\begin{verbatim}
head test.txt
head test.txt -n 4
head test.txt -c 10
\end{verbatim}
