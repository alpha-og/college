\subsection{Command 50: cut} 
\textbf{Purpose}
\begin{flushleft}
 cut out selected portions of each line of a file
\end{flushleft}
\textbf{Usage}
\begin{verbatim}
cut --characters=3 testMod.txt
\end{verbatim}
