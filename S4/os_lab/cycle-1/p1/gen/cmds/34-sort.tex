\subsection{Command 34: sort} 
\textbf{Purpose}
\begin{flushleft}
 sort or merge records (lines) of text and binary files
\end{flushleft}
\textbf{Usage}
\begin{verbatim}
sort test.txt
sort -r test.txt
\end{verbatim}
