\subsection{Command 16: chmod} 
\textbf{Purpose}
\begin{flushleft}
 change file modes or Access Control Lists
\end{flushleft}
\textbf{Usage}
\begin{verbatim}
chmod +x test.sh
chmod +r test.sh
chmod +w test.sh
\end{verbatim}
