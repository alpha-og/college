\subsection{Command 37: uniq} 
\textbf{Purpose}
\begin{flushleft}
 report or filter out repeated lines in a file
\end{flushleft}
\textbf{Usage}
\begin{verbatim}
uniq test.txt
uniq test.txt -c
uniq test.txt -d
\end{verbatim}
