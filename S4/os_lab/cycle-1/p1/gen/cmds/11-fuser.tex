\subsection{Command 11: fuser} 
\textbf{Purpose}
\begin{flushleft}
       fuser - list process IDs of all processes that have one or more files
\end{flushleft}
\textbf{Usage}
\begin{verbatim}
fuser -l
\end{verbatim}
