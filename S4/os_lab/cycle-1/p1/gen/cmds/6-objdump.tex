\subsection{Command 6: objdump} 
\textbf{Purpose}
\begin{flushleft}
       llvm-objdump - LLVM's object file dumper
\end{flushleft}
\textbf{Usage}
\begin{verbatim}
objdump -f ./a.out
objdump -S ./a.out
\end{verbatim}
