\subsection{Command 31: tail} 
\textbf{Purpose}
\begin{flushleft}
 display the last part of a file
\end{flushleft}
\textbf{Usage}
\begin{verbatim}
tail test.txt
tail test.txt -n 4
tail test.txt -c 10
\end{verbatim}
